
import java.util.Random;

public class RandomizerUtil {
	private Random rand = new Random();

	public int rndZeroToNum(int num) {
		// Generate random integers in range 0 to num
		return rand.nextInt(num);
	}

	public String terneryExp(Boolean b1, String exp1, String exp2) {
		return b1 ? exp1 : exp2;
	}
}
