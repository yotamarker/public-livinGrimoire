import java.text.DecimalFormat;

public class Confidence {
	// machine learning
	private Double yes = 0.0;
	private Double total = 0.0;

	public void input(Boolean b1) {
		if (b1) {
			yes++;
		}
		total++;
	}

	public Double getConfidencePercent() {
		return Double.parseDouble(new DecimalFormat("##.##").format((yes / total) * 100));
	}

	public void reset() {
		yes = 0.0;
		total = 0.0;
	}
}
