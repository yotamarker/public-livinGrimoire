import java.util.ArrayList;
import java.util.Arrays;

public class InputFilterEarContains extends InputFilter {
    // returns inputs containing key inputs without the key inputs
    private ArrayList<String> keyInputs;

    public InputFilterEarContains(String... keyInputs) {
        super();
        this.keyInputs = new ArrayList<>(Arrays.asList(keyInputs));
    }

    @Override
    public String filterInput(String ear, String skin, String eye) {
        for (String str1 : keyInputs) {
            if (ear.contains(str1)) {
                return ear.replace(str1, "");
            }
        }
        return "";
    }
}
