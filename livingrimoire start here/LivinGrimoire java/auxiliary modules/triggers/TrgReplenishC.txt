public class TrgReplenishC {
	private int MAXCOUNTER = 1;
	private int counter = MAXCOUNTER;
	private String replenishTime = "14:00";
	private PlayGround playGround = new PlayGround();
	public Boolean trigger() {
		if(counter > 0) {counter--;return true;}
		return false;
	}
	public void reset() {
		if(playGround.getCurrentTimeStamp().equals(replenishTime)) {counter = MAXCOUNTER;}
	}
}
