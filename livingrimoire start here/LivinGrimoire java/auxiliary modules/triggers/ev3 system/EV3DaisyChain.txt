import java.util.ArrayList;

public class EV3DaisyChain {
	// this class connects several logic gates triggers together
	ArrayList<TrGEV3> trgGates = new ArrayList<>();

	public EV3DaisyChain(TrGEV3... gates) {
		for (int i = 0; i < gates.length; i++) {
			trgGates.add(gates[i]);
		}
	}

	public void reset() {
		trgGates.forEach((n) -> n.reset());
	}

	public void input(String ear, String skin, String eye) {
		trgGates.forEach((n) -> n.input(ear, skin, eye));
	}

	public Boolean trigger() {
		Boolean result = true;
		for (TrGEV3 i : trgGates) {
			result &= i.trigger();
		}
		return result;
	}
}
