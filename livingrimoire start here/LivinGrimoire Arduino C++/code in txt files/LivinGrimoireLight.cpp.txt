// ListOfSkills class
void ListOfSkills::append(Skill* item) 
{
  if (length < 16) data[length++] = item;
}
void ListOfSkills::remove(byte index) 
{
  if (index >= length) return;
 memmove(&data[index], &data[index+1], length -
index - 1);
 length--;
}
