*********
Public Function Sum1(ByVal numbers() As Integer) As Integer
    Dim total As Integer = 0
    For Each number As Integer In numbers
        total += number
    Next
    Return total
End Function

Dim test As Integer = Sum1(New Integer() {1, 2, 3})
