package com.yotamarker.lgkotlin1;

public class Coordinates {
    private Double x = 0.000;
    private Double y = 0.000;

    public Coordinates(Double x, Double y) {
        super();
        this.x = x;
        this.y = y;
    }

    public Double getX() {
        return x;
    }

    public void setX(Double x) {
        this.x = x;
    }

    public Double getY() {
        return y;
    }

    public void setY(Double y) {
        this.y = y;
    }

}
